// ============================================================ // SCRIPT COMPLET – MEMO SCAN (coûts corrigés) // Scan = 0,001 USD / Exercice = 0,003 USD // ============================================================ const SHEET_ID = 'TON_ID_ICI'; // ← Remplace par l'ID de ton Google Sheets const SHEET_NAME = 'Inscriptions'; const COUT_SCAN = 0.001; const COUT_EXERCICE = 0.003; function doPost(e) { const action = e.parameter.action; const email = e.parameter.email; const password = e.parameter.password; const exercice = e.parameter.exercice === 'true'; const today = new Date(); const todayStr = today.toISOString().slice(0,10); const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); let row = -1; for (let i = 1; i < data.length; i++) { if (data[i][1] === email) { row = i; break; } } // === INSCRIPTION === if (action === 'register') { if (row !== -1) return ContentService.createTextOutput('EXISTANT'); const hashed = Utilities.computeDigest(Utilities.DigestAlgorithm.SHA_256, password) .map(b => ('0' + (b & 0xFF).toString(16)).slice(-2)) .join(''); // A=Timestamp, B=Email, C=PasswordHash, N=Status (ACTIF) sheet.appendRow([today, email, hashed, '', '', '', '', 0, 0, 0, 0, 0, 14, 'ACTIF']); return ContentService.createTextOutput('OK'); } if (row === -1) return ContentService.createTextOutput('UTILISATEUR_INCONNU'); // === ACTIVITÉ (scan ou exercice) === if (action === 'activity') { const status = data[row][13]; let scans = data[row][7] || 0; let exos = data[row][8] || 0; let coutScans = data[row][9] || 0; let coutExos = data[row][10] || 0; let totalCout = data[row][11] || 0; let joursRestants = data[row][12] || 14; let lastActive = data[row][14]; if (status === 'ACTIF') { if (exercice) { exos++; coutExos = exos * COUT_EXERCICE; } else { scans++; coutScans = scans * COUT_SCAN; } totalCout = coutScans + coutExos; const signupDate = new Date(data[row][0]); const diffDays = Math.floor((today - signupDate) / (1000 * 60 * 60 * 24)); joursRestants = Math.max(14 - diffDays, 0); sheet.getRange(row+1, 8).setValue(scans); sheet.getRange(row+1, 9).setValue(exos); sheet.getRange(row+1, 10).setValue(coutScans); sheet.getRange(row+1, 11).setValue(coutExos); sheet.getRange(row+1, 12).setValue(totalCout); sheet.getRange(row+1, 13).setValue(joursRestants); sheet.getRange(row+1, 15).setValue(todayStr); // LastActiveDate (colonne O) // (optionnel) déclencher avis Google Business après 3 jours actifs } return ContentService.createTextOutput('OK'); } // === WEBHOOK WISE (paiement) === if (action === 'webhook') { const reference = e.parameter.reference || ''; sheet.getRange(row+1, 14).setValue('PAYÉ'); sheet.getRange(row+1, 16).setValue(today); // DatePaiement sheet.getRange(row+1, 17).setValue(reference); // ReferenceWise return ContentService.createTextOutput('OK'); } return ContentService.createTextOutput('ERREUR'); } function doGet() { return ContentService.createTextOutput('Service actif'); }
